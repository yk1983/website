HashMapEx3.java

전화번호를 저장할 그룹을 만들고 그룹 안에 다시 이름과 전화번호를 저장하는 예제입니다. 이름은 중복이 있을 수 있지만 전화번호는 유일한 번호이어야 합니다.

소스코드

package kinco.java.std;

import java.util.*;

public class HashMapEx3 {
	static HashMap<String, HashMap<?, ?> phoneBook = new HashMap<String, HashMap<?, ?>();

	public static void main(String[] args) {
		addPhoneNo("친구", "이자바", "010-111-1111");
		addPhoneNo("친구", "김자바", "010-222-2222");
		addPhoneNo("친구", "김자바", "010-333-3333");
		addPhoneNo("회사", "김대리", "010-444-4444");
		addPhoneNo("회사", "김대리", "010-555-5555");
		addPhoneNo("회사", "박대리", "010-666-6666");
		addPhoneNo("회사", "이과장", "010-777-7777");
		addPhoneNo("세탁", "010-888-8888");

		printList();
	}

	// 그룹을 추가하는 메서드
	@SuppressWarnings("rawtypes")
	static void addGroup(String groupName) {
		if (!phoneBook.containsKey(groupName)) {
			phoneBook.put(groupName, new HashMap());	
		}
	}

	// 그룹에 전화번호를 추가하는 메서드
	static void addPhoneNo(String groupName, String name, String tel) {
		addGroup(groupName);
		@SuppressWarnings("unchecked")
		HashMap<String, String> group = (HashMap<String, String>) phoneBook.get(groupName);
		group.put(tel, name); // 이름은 중복될 수 있으니 전화번호를 key로 저장한다.
	}

	static void addPhoneNo(String name, String tel) {
		addPhoneNo("기타", name, tel);
	}

	// 전화번호부 전체를 출력하는 메서드
	static void printList() {
		Set<?> set = phoneBook.entrySet();
		Iterator<?> it = set.iterator();

		while (it.hasNext()) {
			@SuppressWarnings("rawtypes")
			Map.Entry e = (Map.Entry) it.next();

			Set<?> subSet = ((HashMap<?, ?>) e.getValue()).entrySet();
			Iterator<?> subIt = subSet.iterator();

			System.out.println(" * " + e.getKey() + "[" + subSet.size() + "]");

			while (subIt.hasNext()) {
				@SuppressWarnings("rawtypes")
				Map.Entry subE = (Map.Entry) subIt.next();
				String telNo = (String) subE.getKey();
				String name = (String) subE.getValue();
				System.out.println(name + " " + telNo);
			}
			System.out.println();
		}
	}
}
