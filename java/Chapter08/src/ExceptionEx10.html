ExceptionEx10.java

메서드의 throws를 명시하고 자신을 호출한 메서드에 예외처리를 전달하는 예제

소스코드

package kinco.java.std;

public class ExceptionEx10 {
	public static void main(String[] args) throws Exception {
		// main 메서드를 호출한 메서드가 없고,
		// 예외처리를 해주지 않았으므로 비정상적 종료
		myMethod1();
	}

	private static void myMethod1() throws Exception{
		// 이 메서드를 호출한 곳에 예외처리를 전달
		myMethod2();
	}

	private static void myMethod2() throws Exception {
		// 예외를 발생
		// 이 메서드를 호출한 곳에 예외처리를 전달
		throw new Exception();
	}
}
