package kinco.java.std;

public class UserExceptionTest {
	public static void main(String[] args) {
		try {
			// 프로그램 설치에 필요한 준비
			startInstall();
			// 파일들을 복사
			copyFiles();
		} catch (SpaceException e) {
			System.out.println("에러 메시지 : " + e.getMessage());
			e.printStackTrace();
			System.out.println("공간을 확보한 후에 다시 설치하시기 바랍니다.");
		} catch (MemoryException me) {
			System.out.println("에러 메시지 : " + me.getMessage());
			me.printStackTrace();
			// Garbage Collection을 수행하여 메모리를 눌려줌
			System.gc();
			System.out.println("다시 설치를 시도하세요.");
		} finally {
			// 프로그램 설치에 사용된 임시파일들을 삭제
			deleteTempFiles();
		}
	}

	/**
		* 프로그램 설치에 필요한 준비를 하는 메서드
		*/
	static void startInstall() throws SpaceException, MemoryException {
		/* 프로그램 설치에 필요한 준비를 하는 코드 */
		// 충분한 설치 공간이 없는 경우
		if (!enoughSpace()) {
			throw new SpaceException("설치할 공간이 부족합니다.");
		}
		// 충분한 메모리가 없는 경우
		if (!enoughMemory()) {
			throw new MemoryException("메모리가 부족합니다.");
		}
	}

	/**
		* 설치에 필요한 파일들을 복사하는 메서드
		*/
	private static void copyFiles() {
		/* 파일들을 복사하는 코드 */
		System.out.println("임시파일 복사");
	}

	/**
		* 프로그램 설치에 사용된 임시파일들을 삭제하는 메서드
		*/
	private static void deleteTempFiles() {
		/* 임시파일들을 삭제하는 코드 */
		System.out.println("임시파일 삭제");
	}

	/**
		* 저장공간 확인 메서드
		* 
		* @return
		*/
	static boolean enoughSpace() {
		// 설치하는데 필요한 공간이 있는지 확인하는 코드를 적는다.
		return false;
	}

	/**
		* 메모리 확인 메서드
		* 
		* @return
		*/
	static boolean enoughMemory() {
		// 설치하는데 필요한 메모리공간이 있는지 확인하는 코드를 적는다.
		return true;
	}
}
