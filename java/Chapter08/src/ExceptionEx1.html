ExceptionEx1.java

예외처리가 되지 않았을 때 랜덤함수를 이용 100을 나누는 프로그램 실행시 랜덤함수의 값이 0이 추출되는 경우 예외적인 상황이 발생하는 예제.

소스코드

package kinco.java.std;

public class ExceptionEx1 {
	public static void main(String[] args) {
		int number = 100;
		int result = 0;
		
		for (int i = 0; i < 10; i++) {
			// 랜덤 값이 0이 추출될 때 예외적인 상황이 발생
			result = number / (int) (Math.random() * 10);
			System.out.println(result);
		}
	}
}
