try-catch-finally 문법

소스코드

package kinco.java.std;

public class ExceptionEx7 {
	public static void main(String[] args) {
		try {
			// 프로그램 설치에 필요한 준비
			startInstall();
			// 설치에 필요한 파일들을 복사
			copyFiles();
//			deleteTempFiles();
		} catch (Exception e) {
			e.printStackTrace();
//			deleteTempFiles();
		} finally {
			// 예외발생여부에 상관없이 실행
			// 프로그램 설치에 사용된 임시파일들을 삭제
			deleteTempFiles();
		}
	}

	/**
	 * 프로그램 설치에 필요한 준비를 하는 메서드
	 */
	private static void startInstall() {
		/* 프로그램 설치에 필요한 준비를 하는 코드 */
		System.out.println("# 프로그램 설치 준비 #");
	}
	
	/**
	 * 설치에 필요한 파일들을 복사하는 메서드
	 */
	private static void copyFiles() {
		/* 파일들을 복사하는 코드  */
		System.out.println("# 임시파일 복사 #");
	}
	
	/**
	 * 프로그램 설치에 사용된 임시파일들을 삭제하는 메서드
	 */
	private static void deleteTempFiles() {
		/* 임시파일들을 삭제하는 코드 */
		System.out.println("# 임시파일 삭제 #");
	}
}
