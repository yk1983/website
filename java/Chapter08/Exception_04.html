사용자정의 예외

사용자정의 예외 만들기

기존의 정의된 예외 클래스 외에 필요에 따라서 프로그래머가 새로운 예외 클래스를 정의하여 사용할 수 있습니다. 이 때 보통은 Exception클래스로부터 상속받는 클래스를 만들게 되지만 필요에 따라서 알맞은 예외 클래스를 선택할 수 있습니다.

사용자 정의 예외

package kinco.java.std;

public class MyException extends Exception {
	// 문자열을 매개변수로 받는 생성자
	public MyException(String msg) {
		// 조상인 Exception클래스의 생성자를 호출
		super(msg);
	}
}

사용자 정의 예외 메서드 추가

package kinco.java.std;

public class MyException extends Exception {
	/**
	 * serialVersionUID 는 직렬화에 사용되는 고유 아이디
	 * 선언하지 않으면 JVM에서 디폴트로 자동 생성
	 * 선언하지 않아도 동작하는데 문제는 없음
	 * JAVA에서는 명시적으로 serialVersionUID를 선언할 것을 권장
	 */
	private static final long serialVersionUID = 1L;
	// 에러코드 값을 저장하기 위한 필드를 추가
	private final int ERR_CODE;
	
	/**
	 * 생성자
	 * @param msg
	 */
	public MyException(String msg) {
		this(msg, 100);  // ERR_CODE를 100(기본값)으로 초기화
	}
	
	/**
	 * 생성자
	 * @param msg
	 * @param errCode
	 */
	public MyException(String msg, int errCode) {
		super(msg);
		ERR_CODE = errCode;
	}
	
	/**
	 * 에러코드를 얻어옴
	 * 이 메서드는 주로 getMessage()와 함께 사용됨
 	 * @return ERR_CODE
	 */
	public int getErrCode() {
		return ERR_CODE;
	}
}
