DataOutputStreamEx3.java

int형 배열 score의 값들을 DataOutputStream을 이용하여 score.dat 파일에 출력하는 예제입니다.
DataInputStreamEx1.java 파일을 이용하여 출력해보도록 하세요.

소스코드

package std.java.chapter15;

import java.io.DataOutputStream;
import java.io.FileOutputStream;
import java.io.IOException;

public class DataOutputStreamEx3 {
	public static void main(String args[]) {
		int[] score = { 100, 90, 95, 85, 50 };

		try {
			FileOutputStream fos = new FileOutputStream("src/std/java/chapter15/score.dat");
			DataOutputStream dos = new DataOutputStream(fos);

			for (int i = 0; i < score.length; i++) {
				dos.writeInt(score[i]);
			}

			dos.close();
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
}
