FileCopy.java

FileInputStream과 FileOutputStream을 사용해서 FileCopy.java파일의 내용을 그대로 FileCopy.bak로 복사하는 예제입니다.
read()로 파일을 읽어서, write(int b)로 파일을 출력합니다.
* 텍스트 파일을 다루는 경우에는 FileInputStream/FileOutputStream보다는 문자 기반 스트림인 FileReader/FileWriter를 사용하는 것이 더 좋습니다.

소스코드

package kinco.java.std;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;

public class FileCopy {
	public static void main(String args[]) {
		try {
			FileInputStream fis = new FileInputStream("src/website/java/std/FileCopy.java");
			FileOutputStream fos = new FileOutputStream("src/website/java/std/FileCopy.bak");

			int data = 0;
			while ((data = fis.read()) != -1) {
				fos.write(data); // void write(int b)
			}

			fis.close();
			fos.close();
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
}
