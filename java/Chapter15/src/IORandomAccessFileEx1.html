RandomAccessFileEx1.java

파일에 출력작업이 수행되었을 때 파일 포인터의 위치가 어떻게 달라지는지에 대한 예제입니다. int가 4byte이기 때문에 writeInt()를 호출한 다음 파일 포인터의 위치가 0에서 4로 바뀝니다. 마찬가지로 8byte인 long을 출력하는 wtiteLong()을 호출한 후에는 파일 포인터의 위치가 4에서 12로 변경됩니다.

소스코드

package std.java.chapter15;

import java.io.IOException;
import java.io.RandomAccessFile;

public class RandomAccessFileEx1 {
	private static RandomAccessFile raf;

	public static void main(String[] args) {
		try {
			raf = new RandomAccessFile("src/std/java/chapter15/RandomAccessFileEx1.dat", "rw");
			System.out.println("파일 포인터의 위치: " + raf.getFilePointer());
			raf.writeInt(100);
			System.out.println("파일 포인터의 위치: " + raf.getFilePointer());
			raf.writeLong(100L);
			System.out.println("파일 포인터의 위치: " + raf.getFilePointer());
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
}
