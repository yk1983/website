DataInputStreamEx2.java

DataOutputStreamEx3.java 예제에서 만들어진 score.dat파일을 읽어서 합계를 구한 후 화면에 출력하는 예제입니다.
int의 크기가 4byte이므로 모두 20byte의 데이터가 저장되어 있습니다. (16진수 두 자리는 1byte입니다.)

소스코드

package std.java.chapter15;

import java.io.DataInputStream;
import java.io.EOFException;
import java.io.FileInputStream;
import java.io.IOException;

public class DataInputStreamEx2 {
	public static void main(String args[]) {
		int sum = 0;
		int score = 0;

		FileInputStream fis = null;
		DataInputStream dis = null;

		try {
			fis = new FileInputStream("src/std/java/chapter15/score.dat");
			dis = new DataInputStream(fis);

			while (true) {
				// 데이터를 읽는 메서드는 더 이상 읽을 데이터가 없으면 EOFException을 발생.
				score = dis.readInt();
				System.out.println(score);
				sum += score;
			}
		} catch (EOFException e) {
			System.out.println("점수의 총합은 " + sum + "입니다.");
		} catch (IOException ie) {
			ie.printStackTrace();
		} finally {
			try {
				// 참조 변수 dis가 null일 때 close()메서드를 호출하면,
				// NullPointerExcetoion이 발생.
				if (dis != null) {
					dis.close();
				}
			} catch (IOException ie) {
				ie.printStackTrace();
			}
		}
	}
}
