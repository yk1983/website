StandardIOEx1.java

실행시 화면에 커서가 기다리고 있으며 "HELLO"를 입력하고 Ctrl + Z를 누르거나 Enter를 누르면 입력한 문자들이 출력되고 프로그램이 종료됩니다.

소스코드

package std.java.chapter15;

import java.io.IOException;

public class StandardIOEx1 {
	public static void main(String[] args) {
		try {
			int input = 0;

			// 사용자의 입력을 기다림
			while ((input = System.in.read()) != -1) {
				System.out.println("input : " + input + ", (char)input :" + (char) input);
			}
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
}
