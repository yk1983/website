FileEx8.java

재귀호출을 이용해서 지정된 디렉토리와 하위 디렉토리에 있는 파일 중에서 지정된 확장자를 가진 파일을 delete()를 호출해서 삭제하는 예제입니다.
삭제 메서드는 해당 파일을 삭제하고 성공하면 true를 실패하면 false를 반환합니다.

소스코드

package chapter15;

import java.io.File;

public class FileEx8 {
	static int deletedFiles = 0;

	public static void main(String[] args) {

		String extension = "bak";
		
		String currDir = System.getProperty("user.dir");

		File dir = new File(currDir);
		String ext = "." + extension;

		delete(dir, ext);
		System.out.println(deletedFiles + "개의 파일이 삭제되었습니다.");
	}

	public static void delete(File dir, String ext) {
		File[] files = dir.listFiles();

		for (int i = 0; i < files.length; i++) {
			if (files[i].isDirectory()) {
				delete(files[i], ext);
			} else {
				String filename = files[i].getAbsolutePath();

				if (filename.endsWith(ext)) {
					System.out.print(filename);
					if (files[i].delete()) {
						System.out.println(" - 삭제 성공");
						deletedFiles++;
					} else {
						System.out.println(" - 삭제 실패");
					}
				}
			}
		}
	}
}
