RandomAccessFileEx3.java

RandomAccessFileEx2.java를 실행한 후 생성된 score2.dat파일에서 국어과목 점수만을 합계를 내는 예제입니다.

소스코드

package std.java.chapter15;

import java.io.EOFException;
import java.io.IOException;
import java.io.RandomAccessFile;

public class RandomAccessFileEx3 {
	private static RandomAccessFile raf;

	public static void main(String args[]) {
		int sum = 0;

		try {
			raf = new RandomAccessFile("src/std/java/chapter15/score2.dat", "r");
			int i = 4;

			while (true) {
				raf.seek(i);
				sum += raf.readInt();
				i += 16;
			}
		} catch (EOFException e) {
			System.out.println("sum : " + sum);
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
}
