SequenceInputStreamEx.java

3개의 ByteArrayInputStream을 Vector와 SequenceInputStream을 이용해서 하나의 입력스트림처럼 다룹니다.
Vector에 저장된 순서대로 입력되므로 순서에 주의하도록 하세요.

소스코드

package std.java.chapter15;

import java.io.ByteArrayInputStream;
import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.SequenceInputStream;
import java.util.Arrays;
import java.util.Vector;

public class SequenceInputStreamEx {
	public static void main(String[] args) {
		byte[] arr1 = { 0, 1, 2 };
		byte[] arr2 = { 3, 4, 5 };
		byte[] arr3 = { 6, 7, 8 };
		byte[] outSrc = null;

		Vector<ByteArrayInputStream> v = new Vector<ByteArrayInputStream>();
		v.add(new ByteArrayInputStream(arr1));
		v.add(new ByteArrayInputStream(arr2));
		v.add(new ByteArrayInputStream(arr3));

		SequenceInputStream input = new SequenceInputStream(v.elements());
		ByteArrayOutputStream output = new ByteArrayOutputStream();

		int data = 0;

		try {
			while ((data = input.read()) != -1) {
				// void write(int b)
				output.write(data);
			}
		} catch (IOException e) {
			System.out.println(e.getMessage());
		}

		outSrc = output.toByteArray();

		System.out.println("Input Source1 : " + Arrays.toString(arr1));
		System.out.println("Input Source2 : " + Arrays.toString(arr2));
		System.out.println("Input Source3 : " + Arrays.toString(arr3));
		System.out.println("Output Source : " + Arrays.toString(outSrc));
	}
}
Input Source1 : [0, 1, 2]
Input Source2 : [3, 4, 5]
Input Source3 : [6, 7, 8]
Output Source : [0, 1, 2, 3, 4, 5, 6, 7, 8]
