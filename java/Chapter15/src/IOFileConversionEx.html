FileConversionEx.java

파일의 공백을 모두 없애는 예제입니다.
FileConversionEx.java을 읽어서 공백을 없애고 conversion.txt파일을 생성하여 씁니다.

소스코드

package kinco.java.std;

import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class FileConversionEx {
	public static void main(String args[]) {
		try {
			FileReader fr = new FileReader("src/kinco/java/std/FileConversionEx.java");
			FileWriter fw = new FileWriter("src/kinco/java/std/conversion.txt");

			int data = 0;
			while ((data = fr.read()) != -1) {
				if (data != '\t' && data != '\n' && data != ' ' && data != '\r')
					fw.write(data);
			}

			fr.close();
			fw.close();
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
}
