ThreadPriorityEx

th1과 th2 모두 main에서드에서 생성하였기 때문에 main메서드를 실행하는 스레드의 우선순위인 5를 상속받았습니다. 그 다음에 th2.setPriority(7)로 th2의 우선순위를 7로 변경한 다음에 start()를 호출해서 스레드를 실행시키는 예제입니다.

ThreadPriorityEx01.java

package chapter13;

public class ThreadPriorityEx01 {
	public static void main(String[] args) {
		ThreadPriorityA th1 = new ThreadPriorityA();
		ThreadPriorityB th2 = new ThreadPriorityB();

		th2.setPriority(7);

		System.out.println("Priority of th1(-) : " + th1.getPriority() );
		System.out.println("Priority of th2(|) : " + th2.getPriority() );
		th1.start();
		th2.start();
	}
}

ThreadPriorityA.java

package chapter13;

public class ThreadPriorityA extends Thread {
	public void run() {
		for(int i=0; i <300; i++) {
			System.out.print("-");
			for(int x=0; x <10000000; x++);
		}
	}
}

ThreadPriorityB.java

package chapter13;

public class ThreadPriorityB extends Thread {
	public void run() {
		for(int i=0; i < 300; i++) {
			System.out.print("|");
			for(int x=0; x < 10000000; x++);
		}
	}
}
