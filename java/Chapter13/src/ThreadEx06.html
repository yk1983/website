ThreadEx06

suspend()와 resume(),stop()의 사용법을 보여주는 예제입니다. sleep(2000)은 스레드를 2초간 멈추게 하지만, 2초 후에 실행상태가 아닌 실행대기상태가 됩니다.

ThreadEx06.java

package chapter13;

public class ThreadEx06 {
	public static void main(String[] args) throws Exception {
		RunImplEx06 r = new RunImplEx06();
		Thread th1 = new Thread(r, "*");
		Thread th2 = new Thread(r, "**");
		Thread th3 = new Thread(r, "***");

		th1.start();
		th2.start();
		th3.start();

		try {
			Thread.sleep(2000);
			th1.suspend(); // 쓰레드 th1을 잠시 중단시킨다.
			Thread.sleep(2000);
			th2.suspend();
			Thread.sleep(3000);
			th1.resume(); // 쓰레드 th1이 다시 동작하도록 한다.
			Thread.sleep(3000);
			th1.stop(); // 쓰레드 th1을 강제종료시킨다.
			th2.stop();
			Thread.sleep(2000);
			th3.stop();
		} catch (InterruptedException e) {
		}
	}
}

RunImplEx06.java

package chapter13;

public class RunImplEx06 implements Runnable {
	public void run() {
		while (true) {
			System.out.println(Thread.currentThread().getName());
			try {
				Thread.sleep(1000);
			} catch (InterruptedException e) {
			}
		}
	}
}
